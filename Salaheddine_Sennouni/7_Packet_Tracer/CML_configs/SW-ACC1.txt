! ============================================================
! SW-ACC1 - Access switch Airline A (Cisco IOSvL2 in CML)
! Rol: VLANs 110-112 voor Airline A + Port Security
! Bron: PT v4 + redundante uplinks naar SW-DIST en SW-DIST2
! ============================================================

enable
configure terminal

hostname SW-ACC1
no ip domain-lookup
ip domain-name odisee.local

!
! --- VLANs (PT v4 namen) ---
!
vlan 110
 name R1_A_OFFICE
vlan 111
 name R1_A_CHECKIN
vlan 112
 name R1_A_GATES
vlan 130
 name R1_VOIP
vlan 131
 name R1_IOT
vlan 132
 name R1_PRINTERS
vlan 140
 name R1_WIFI_STAFF
vlan 141
 name R1_WIFI_GUEST
vlan 190
 name R1_NET_MGMT
exit

!
! --- STP - PortFast en BPDUguard standaard op alle access ---
!
spanning-tree mode rapid-pvst
spanning-tree portfast default
spanning-tree portfast bpduguard default

!
! --- Uplink trunks (primary + backup) ---
!
interface GigabitEthernet0/0
 description UPLINK-TO-SW-DIST
 switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk allowed vlan 110,111,112,130,131,132,140,141,190
 spanning-tree portfast disable
 no shutdown

interface GigabitEthernet0/1
 description UPLINK-TO-SW-DIST2-BACKUP
 switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk allowed vlan 110,111,112,130,131,132,140,141,190
 spanning-tree portfast disable
 no shutdown

!
! --- Access poorten Airline A ---
!
interface GigabitEthernet0/2
 description PC-A1-OFFICE
 switchport mode access
 switchport access vlan 110
 switchport voice vlan 130
 switchport port-security
 switchport port-security maximum 3
 switchport port-security mac-address sticky
 switchport port-security violation restrict
 switchport port-security aging time 10
 switchport port-security aging type inactivity
 no shutdown

interface GigabitEthernet0/3
 description PC-A2-CHECKIN
 switchport mode access
 switchport access vlan 111
 switchport voice vlan 130
 switchport port-security
 switchport port-security maximum 3
 switchport port-security mac-address sticky
 switchport port-security violation restrict
 no shutdown

interface GigabitEthernet0/4
 description SCANNER-A-GATES
 switchport mode access
 switchport access vlan 112
 switchport port-security
 switchport port-security maximum 1
 switchport port-security mac-address sticky
 switchport port-security violation restrict
 no shutdown

!
! --- Management SVI + default-gateway ---
!
interface Vlan190
 description MANAGEMENT
 ip address 10.1.190.11 255.255.255.0
 no shutdown

ip default-gateway 10.1.190.1

!
! --- SSH v2 (CML vereist 2048 modulus) ---
!
crypto key generate rsa modulus 2048
ip ssh version 2

username admin privilege 15 secret Cisco123!

line vty 0 4
 transport input ssh
 login local
line vty 5 15
 transport input ssh
 login local

end
write memory
