! ============================================================
! SW-ACC2 - Access switch Airline B (Cisco IOSvL2 in CML)
! Rol: VLANs 120-122 voor Airline B + Port Security
! ============================================================

enable
configure terminal

hostname SW-ACC2
no ip domain-lookup
ip domain-name odisee.local

vlan 120
 name R1_B_OFFICE
vlan 121
 name R1_B_CHECKIN
vlan 122
 name R1_B_GATES
vlan 130
 name R1_VOIP
vlan 131
 name R1_IOT
vlan 132
 name R1_PRINTERS
vlan 140
 name R1_WIFI_STAFF
vlan 141
 name R1_WIFI_GUEST
vlan 190
 name R1_NET_MGMT
exit

spanning-tree mode rapid-pvst
spanning-tree portfast default
spanning-tree portfast bpduguard default

interface GigabitEthernet0/0
 description UPLINK-TO-SW-DIST
 switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk allowed vlan 120,121,122,130,131,132,140,141,190
 spanning-tree portfast disable
 no shutdown

interface GigabitEthernet0/1
 description UPLINK-TO-SW-DIST2-BACKUP
 switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk allowed vlan 120,121,122,130,131,132,140,141,190
 spanning-tree portfast disable
 no shutdown

interface GigabitEthernet0/2
 description PC-B1-OFFICE
 switchport mode access
 switchport access vlan 120
 switchport voice vlan 130
 switchport port-security
 switchport port-security maximum 3
 switchport port-security mac-address sticky
 switchport port-security violation restrict
 switchport port-security aging time 10
 switchport port-security aging type inactivity
 no shutdown

interface GigabitEthernet0/3
 description PC-B2-CHECKIN
 switchport mode access
 switchport access vlan 121
 switchport voice vlan 130
 switchport port-security
 switchport port-security maximum 3
 switchport port-security mac-address sticky
 switchport port-security violation restrict
 no shutdown

interface GigabitEthernet0/4
 description SCANNER-B-GATES
 switchport mode access
 switchport access vlan 122
 switchport port-security
 switchport port-security maximum 1
 switchport port-security mac-address sticky
 switchport port-security violation restrict
 no shutdown

interface Vlan190
 description MANAGEMENT
 ip address 10.1.190.12 255.255.255.0
 no shutdown

ip default-gateway 10.1.190.1

crypto key generate rsa modulus 2048
ip ssh version 2

username admin privilege 15 secret Cisco123!

line vty 0 4
 transport input ssh
 login local
line vty 5 15
 transport input ssh
 login local

end
write memory
